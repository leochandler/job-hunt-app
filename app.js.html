import React, { useState, useEffect } from 'react';

import { Plus, Trash2, Circle, Heart, Zap, Coffee, TrendingUp, Calendar, Star, Building2, Users, Target, Edit2, CheckSquare, X, Clock, Play, Pause, RotateCcw, BookOpen, Award, ListTodo, Lightbulb, ArrowRight, DollarSign, MessageCircle, Activity, Smile, PhoneCall, Sun, Download } from 'lucide-react';


// ==================== INITIAL DATA ====================

const INITIAL_STATE = {

  applications: [],

  wins: [],

  moodEntries: [],

  targetCompanies: [],

  networkingContacts: [],

  jobCriteria: {

    mustHaves: ['Remote work', 'Health insurance'],

    niceToHaves: ['Flexible hours', '401k matching'],

    salaryMin: '',

    salaryMax: '',

    dealBreakers: []

  },

  skillsGap: {

    have: ['JavaScript', 'React', 'Node.js'],

    learning: [],

    needed: []

  },

  weeklyGoals: {

    currentWeek: {

      goals: [],

      completed: [],

      reflection: ''

    },

    history: []

  },

  masterTodo: [],

  businessIdeas: [],

  todayFocusTasks: [],

  dailyWellness: [],

  pomodoro: {

    workMinutes: 25,

    breakMinutes: 5,

    sessionsToday: 0,

    applicationsThisSession: 0

  },

  dailyStreak: 0,

  lastActiveDate: new Date().toDateString(),

  totalTasksCompleted: 0,

  wellnessStreak: 0

};


// ==================== LOCAL STORAGE HELPERS ====================

const STORAGE_KEY = 'job-hunt-wellness-data';


const saveToStorage = (data) => {

  try {

    localStorage.setItem(STORAGE_KEY, JSON.stringify(data));

  } catch (error) {

    console.error('Failed to save to localStorage:', error);

  }

};


const loadFromStorage = () => {

  try {

    const stored = localStorage.getItem(STORAGE_KEY);

    return stored ? JSON.parse(stored) : INITIAL_STATE;

  } catch (error) {

    console.error('Failed to load from localStorage:', error);

    return INITIAL_STATE;

  }

};


// ==================== MAIN APP ====================

export default function JobHuntApp() {

  // Load initial state from localStorage

  const [state, setState] = useState(() => loadFromStorage());

  const [activeTab, setActiveTab] = useState('focus');

  const [showInstallPrompt, setShowInstallPrompt] = useState(false);

  const [deferredPrompt, setDeferredPrompt] = useState(null);


  // Save to localStorage whenever state changes

  useEffect(() => {

    saveToStorage(state);

  }, [state]);


  // PWA Install Prompt Handler

  useEffect(() => {

    const handler = (e) => {

      e.preventDefault();

      setDeferredPrompt(e);

      setShowInstallPrompt(true);

    };

    

    window.addEventListener('beforeinstallprompt', handler);

    

    return () => window.removeEventListener('beforeinstallprompt', handler);

  }, []);


  const handleInstall = async () => {

    if (!deferredPrompt) return;

    

    deferredPrompt.prompt();

    const { outcome } = await deferredPrompt.userChoice;

    

    if (outcome === 'accepted') {

      setShowInstallPrompt(false);

    }

    

    setDeferredPrompt(null);

  };


  // Helper to update state immutably

  const updateState = (updates) => {

    setState(prev => ({ ...prev, ...updates }));

  };


  const updateStreak = () => {

    const today = new Date().toDateString();

    if (state.lastActiveDate !== today) {

      updateState({ 

        dailyStreak: state.dailyStreak + 1, 

        lastActiveDate: today 

      });

    }

  };


  const completeTask = (taskId, taskText) => {

    updateState({

      todayFocusTasks: state.todayFocusTasks.filter(t => t.id !== taskId),

      totalTasksCompleted: state.totalTasksCompleted + 1,

      wins: [{

        id: Date.now(),

        text: `Completed: ${taskText}`,

        category: 'general',

        date: new Date().toISOString()

      }, ...state.wins]

    });

    updateStreak();

  };


  const addApplication = (company, position, status = 'interested', energy = 'medium', criteriaMatch = null) => {

    updateState({

      applications: [...state.applications, {

        id: Date.now(),

        company,

        position,

        status,

        energy,

        dateAdded: new Date().toISOString(),

        notes: '',

        criteriaMatch

      }],

      pomodoro: {

        ...state.pomodoro,

        applicationsThisSession: state.pomodoro.applicationsThisSession + 1

      }

    });

    updateStreak();

  };


  const updateAppStatus = (id, newStatus) => {

    updateState({

      applications: state.applications.map(app => 

        app.id === id ? { ...app, status: newStatus } : app

      )

    });

  };


  const deleteApp = (id) => {

    updateState({

      applications: state.applications.filter(app => app.id !== id)

    });

  };


  const addWin = (text, category = 'general') => {

    updateState({

      wins: [{

        id: Date.now(),

        text,

        category,

        date: new Date().toISOString()

      }, ...state.wins]

    });

    updateStreak();

  };


  const addMoodEntry = (mood, note = '') => {

    updateState({

      moodEntries: [{

        id: Date.now(),

        mood,

        note,

        date: new Date().toISOString()

      }, ...state.moodEntries]

    });

  };


  const addMasterTodo = (text, priority = 'medium', category = 'general') => {

    updateState({

      masterTodo: [...state.masterTodo, {

        id: Date.now(),

        text,

        priority,

        category,

        completed: false,

        dateAdded: new Date().toISOString()

      }]

    });

    updateStreak();

  };


  const toggleMasterTodo = (id) => {

    const todo = state.masterTodo.find(t => t.id === id);

    if (todo && !todo.completed) {

      updateState({

        totalTasksCompleted: state.totalTasksCompleted + 1,

        wins: [{

          id: Date.now(),

          text: `Completed: ${todo.text}`,

          category: 'general',

          date: new Date().toISOString()

        }, ...state.wins],

        masterTodo: state.masterTodo.map(t =>

          t.id === id ? { ...t, completed: true } : t

        )

      });

    } else {

      updateState({

        masterTodo: state.masterTodo.map(t =>

          t.id === id ? { ...t, completed: false } : t

        )

      });

    }

  };


  const deleteMasterTodo = (id) => {

    updateState({

      masterTodo: state.masterTodo.filter(t => t.id !== id)

    });

  };


  const addBusinessIdea = (title, description) => {

    updateState({

      businessIdeas: [...state.businessIdeas, {

        id: Date.now(),

        title,

        description,

        status: 'idea',

        nextAction: '',

        suggestedActions: [],

        customerConversations: 0,

        revenue: 0,

        dateAdded: new Date().toISOString(),

        actions: []

      }]

    });

  };


  const updateBusinessIdea = (id, updates) => {

    updateState({

      businessIdeas: state.businessIdeas.map(idea =>

        idea.id === id ? { ...idea, ...updates } : idea

      )

    });

  };


  const completeBusinessAction = (ideaId, action) => {

    const idea = state.businessIdeas.find(i => i.id === ideaId);

    if (idea) {

      updateBusinessIdea(ideaId, {

        actions: [...(idea.actions || []), { 

          text: action, 

          completed: new Date().toISOString() 

        }],

        nextAction: ''

      });

      addWin(`Business progress: ${action}`, 'general');

      updateState({ totalTasksCompleted: state.totalTasksCompleted + 1 });

    }

  };


  const addWellnessActivity = (activity, category) => {

    const today = new Date().toDateString();

    const todayActivities = state.dailyWellness.filter(

      a => new Date(a.date).toDateString() === today

    );

    

    updateState({

      dailyWellness: [{

        id: Date.now(),

        activity,

        category,

        date: new Date().toISOString()

      }, ...state.dailyWellness],

      wellnessStreak: todayActivities.length === 0 ? state.wellnessStreak + 1 : state.wellnessStreak

    });

    

    addWin(`Wellness: ${activity}`, 'general');

    updateStreak();

  };


  const stats = {

    total: state.applications.length,

    interested: state.applications.filter(a => a.status === 'interested').length,

    applied: state.applications.filter(a => a.status === 'applied').length,

    interview: state.applications.filter(a => a.status === 'interview').length,

    offer: state.applications.filter(a => a.status === 'offer').length,

    rejected: state.applications.filter(a => a.status === 'rejected').length

  };


  const getTodaysFocus = () => {

    const tasks = [...state.todayFocusTasks];

    const today = new Date().toDateString();

    

    const todayWellness = state.dailyWellness.filter(

      a => new Date(a.date).toDateString() === today

    );

    

    if (todayWellness.length === 0 && tasks.length < 6) {

      tasks.push({ 

        id: 'wellness-reminder', 

        text: '⚡ Take a wellness break - exercise, meditate, or call a friend', 

        energy: 'low', 

        source: 'wellness' 

      });

    }

    

    const highPriorityTodos = state.masterTodo

      .filter(t => !t.completed && t.priority === 'high')

      .slice(0, 2);

    

    highPriorityTodos.forEach(todo => {

      if (!tasks.find(t => t.sourceId === todo.id)) {

        tasks.push({

          id: `todo-${todo.id}`,

          sourceId: todo.id,

          text: todo.text,

          energy: 'medium',

          source: 'todo'

        });

      }

    });

    

    if (stats.interested > 0 && tasks.length < 5) {

      tasks.push({ 

        id: 'ai-1', 

        text: 'Apply to 1-2 positions you saved', 

        energy: 'high', 

        source: 'ai' 

      });

    }

    

    state.businessIdeas

      .filter(idea => idea.nextAction && idea.status !== 'paused')

      .forEach(idea => {

        if (tasks.length < 6 && !tasks.find(t => t.sourceId === idea.id)) {

          tasks.push({

            id: `biz-${idea.id}`,

            sourceId: idea.id,

            text: `[${idea.title}] ${idea.nextAction}`,

            energy: 'low',

            source: 'business'

          });

        }

      });

    

    return tasks.slice(0, 6);

  };


  return (

    <div className="min-h-screen bg-gradient-to-br from-blue-50 via-purple-50 to-pink-50 p-4">

      <div className="max-w-6xl mx-auto">

        

        {/* PWA Install Banner */}

        {showInstallPrompt && (

          <div className="mb-4 bg-gradient-to-r from-purple-500 to-pink-500 text-white rounded-lg p-4 shadow-lg">

            <div className="flex items-center justify-between">

              <div className="flex items-center gap-3">

                <Download className="w-6 h-6" />

                <div>

                  <p className="font-semibold">Install Job Hunt Wellness Tracker</p>

                  <p className="text-sm opacity-90">Add to home screen for quick access & offline use</p>

                </div>

              </div>

              <div className="flex gap-2">

                <button

                  onClick={handleInstall}

                  className="bg-white text-purple-600 px-4 py-2 rounded-lg font-medium hover:bg-gray-100"

                >

                  Install

                </button>

                <button

                  onClick={() => setShowInstallPrompt(false)}

                  className="text-white hover:bg-white hover:bg-opacity-20 px-4 py-2 rounded-lg"

                >

                  <X className="w-5 h-5" />

                </button>

              </div>

            </div>

          </div>

        )}


        <header className="text-center mb-8 pt-6">

          <h1 className="text-4xl font-bold text-gray-800 mb-2">Job Hunt Wellness Tracker</h1>

          <p className="text-gray-600">Focus on progress, not perfection</p>

          <div className="mt-4 flex justify-center gap-4 flex-wrap">

            <div className="inline-flex items-center gap-2 bg-white px-4 py-2 rounded-full shadow-sm">

              <TrendingUp className="w-5 h-5 text-green-500" />

              <span className="font-semibold text-gray-700">{state.dailyStreak} day streak</span>

            </div>

            <div className="inline-flex items-center gap-2 bg-white px-4 py-2 rounded-full shadow-sm">

              <CheckSquare className="w-5 h-5 text-purple-500" />

              <span className="font-semibold text-gray-700">{state.totalTasksCompleted} tasks done</span>

            </div>

            <div className="inline-flex items-center gap-2 bg-white px-4 py-2 rounded-full shadow-sm">

              <Heart className="w-5 h-5 text-pink-500" />

              <span className="font-semibold text-gray-700">{state.wellnessStreak} wellness streak</span>

            </div>

          </div>

        </header>


        <div className="grid grid-cols-2 md:grid-cols-6 gap-3 mb-8">

          <StatCard label="Interested" count={stats.interested} color="blue" />

          <StatCard label="Applied" count={stats.applied} color="purple" />

          <StatCard label="Interview" count={stats.interview} color="yellow" />

          <StatCard label="Offer" count={stats.offer} color="green" />

          <StatCard label="Targets" count={state.targetCompanies.length} color="orange" />

          <StatCard label="Network" count={state.networkingContacts.length} color="teal" />

        </div>


        <div className="flex gap-2 mb-6 overflow-x-auto pb-2">

          <TabButton active={activeTab === 'focus'} onClick={() => setActiveTab('focus')}>

            <Star className="w-4 h-4" /> Focus

          </TabButton>

          <TabButton active={activeTab === 'wellness'} onClick={() => setActiveTab('wellness')}>

            <Activity className="w-4 h-4" /> Wellness

          </TabButton>

          <TabButton active={activeTab === 'todo'} onClick={() => setActiveTab('todo')}>

            <ListTodo className="w-4 h-4" /> Tasks

          </TabButton>

          <TabButton active={activeTab === 'applications'} onClick={() => setActiveTab('applications')}>

            <Calendar className="w-4 h-4" /> Apps

          </TabButton>

          <TabButton active={activeTab === 'business'} onClick={() => setActiveTab('business')}>

            <Lightbulb className="w-4 h-4" /> Business

          </TabButton>

          <TabButton active={activeTab === 'wins'} onClick={() => setActiveTab('wins')}>

            <Heart className="w-4 h-4" /> Wins

          </TabButton>

        </div>


        <div className="bg-white rounded-2xl shadow-lg p-6 min-h-96">

          {activeTab === 'focus' && (

            <FocusTab 

              tasks={getTodaysFocus()} 

              stats={stats}

              completeTask={completeTask}

              todayFocusTasks={state.todayFocusTasks}

              setTodayFocusTasks={(tasks) => updateState({ todayFocusTasks: tasks })}

              updateStreak={updateStreak}

              completeBusinessAction={completeBusinessAction}

              toggleMasterTodo={toggleMasterTodo}

              setActiveTab={setActiveTab}

            />

          )}


          {activeTab === 'wellness' && (

            <WellnessTab

              dailyWellness={state.dailyWellness}

              addWellnessActivity={addWellnessActivity}

              wellnessStreak={state.wellnessStreak}

            />

          )}


          {activeTab === 'todo' && (

            <MasterTodoTab

              masterTodo={state.masterTodo}

              addMasterTodo={addMasterTodo}

              toggleMasterTodo={toggleMasterTodo}

              deleteMasterTodo={deleteMasterTodo}

            />

          )}

          

          {activeTab === 'applications' && (

            <ApplicationsTab 

              applications={state.applications}

              addApplication={addApplication}

              updateAppStatus={updateAppStatus}

              deleteApp={deleteApp}

              jobCriteria={state.jobCriteria}

            />

          )}


          {activeTab === 'business' && (

            <BusinessIdeasTab

              businessIdeas={state.businessIdeas}

              addBusinessIdea={addBusinessIdea}

              updateBusinessIdea={updateBusinessIdea}

              completeBusinessAction={completeBusinessAction}

            />

          )}

          

          {activeTab === 'wins' && (

            <WinsTab wins={state.wins} addWin={addWin} />

          )}

        </div>


        <div className="text-center mt-8 pb-6 text-sm text-gray-500">

          💾 All data saved automatically • Works offline • Remember: You're doing great 💙

        </div>

      </div>

    </div>

  );

}


// ==================== COMPONENTS (Abbreviated for length) ====================

// All components remain the same, just passing through the state


function StatCard({ label, count, color }) {

  const colorClasses = {

    blue: 'bg-blue-100 text-blue-700',

    purple: 'bg-purple-100 text-purple-700',

    yellow: 'bg-yellow-100 text-yellow-700',

    green: 'bg-green-100 text-green-700',

    orange: 'bg-orange-100 text-orange-700',

    teal: 'bg-teal-100 text-teal-700'

  };

  return (

    <div className={`${colorClasses[color]} rounded-xl p-4 text-center shadow-sm`}>

      <div className="text-2xl font-bold">{count}</div>

      <div className="text-xs font-medium mt-1">{label}</div>

    </div>

  );

}


function TabButton({ active, onClick, children }) {

  return (

    <button onClick={onClick} className={`flex items-center gap-2 px-4 py-2 rounded-lg font-medium transition-colors whitespace-nowrap ${active ? 'bg-purple-500 text-white shadow-md' : 'bg-gray-100 text-gray-600 hover:bg-gray-200'}`}>

      {children}

    </button>

  );

}


function FocusTab({ tasks, stats, completeTask, todayFocusTasks, setTodayFocusTasks, updateStreak, completeBusinessAction, toggleMasterTodo, setActiveTab }) {

  const [showAddTask, setShowAddTask] = useState(false);

  const [newTask, setNewTask] = useState('');

  const [taskEnergy, setTaskEnergy] = useState('medium');


  const handleAddTask = () => {

    if (newTask.trim()) {

      setTodayFocusTasks([...todayFocusTasks, {

        id: Date.now(),

        text: newTask.trim(),

        energy: taskEnergy,

        source: 'manual'

      }]);

      setNewTask('');

      setTaskEnergy('medium');

      setShowAddTask(false);

      updateStreak();

    }

  };


  const handleComplete = (task) => {

    if (task.source === 'wellness') {

      setActiveTab('wellness');

      return;

    }

    if (task.source === 'business') {

      const ideaId = task.sourceId;

      const actionText = task.text.includes(']') ? task.text.split('] ')[1] : task.text;

      completeBusinessAction(ideaId, actionText);

    } else if (task.source === 'todo') {

      toggleMasterTodo(task.sourceId);

    }

    completeTask(task.id, task.text);

  };

  

  return (

    <div>

      <div className="flex justify-between items-center mb-4">

        <div>

          <h2 className="text-2xl font-bold text-gray-800">Today's Focus</h2>

          <p className="text-sm text-gray-600">Your most important tasks</p>

        </div>

        <button onClick={() => setShowAddTask(!showAddTask)} className="flex items-center gap-2 bg-purple-500 text-white px-4 py-2 rounded-lg hover:bg-purple-600">

          <Plus className="w-5 h-5" /> Add Task

        </button>

      </div>


      {showAddTask && (

        <div className="bg-purple-50 rounded-lg p-4 mb-6">

          <input type="text" placeholder="What do you need to do today?" value={newTask} onChange={(e) => setNewTask(e.target.value)} onKeyDown={(e) => e.key === 'Enter' && handleAddTask()} className="w-full px-4 py-2 border rounded-lg mb-3" />

          <div className="flex gap-2">

            <select value={taskEnergy} onChange={(e) => setTaskEnergy(e.target.value)} className="px-4 py-2 border rounded-lg">

              <option value="low">Low Energy Required 😌</option>

              <option value="medium">Medium Energy 💪</option>

              <option value="high">High Energy 🔥</option>

            </select>

            <button onClick={handleAddTask} className="bg-purple-500 text-white px-6 py-2 rounded-lg hover:bg-purple-600">Add</button>

          </div>

          <p className="text-xs text-gray-600 mt-2">💡 Match tasks to your current energy level</p>

        </div>

      )}

      

      <div className="space-y-3 mb-8">

        {tasks.length === 0 ? (

          <div className="text-center py-8 text-gray-400"><p>No tasks today. Add one above!</p></div>

        ) : (

          tasks.map((task) => (

            <div key={task.id} className="flex items-start gap-3 p-4 bg-gradient-to-r from-purple-50 to-pink-50 rounded-lg group hover:shadow-md transition-shadow">

              <button onClick={() => handleComplete(task)} className="mt-0.5 flex-shrink-0">

                <Circle className="w-6 h-6 text-purple-400 group-hover:text-purple-600 transition-colors" />

              </button>

              <div className="flex-1">

                <p className="text-gray-800 font-medium">{task.text}</p>

                <div className="flex items-center gap-2 mt-1">

                  <EnergyBadge level={task.energy} />

                  {task.source && <span className="text-xs px-2 py-0.5 rounded bg-gray-100 text-gray-600">{task.source}</span>}

                </div>

              </div>

            </div>

          ))

        )}

      </div>


      <div className="bg-blue-50 rounded-lg p-6 border-l-4 border-blue-400">

        <h3 className="font-semibold text-gray-800 mb-2">💡 Energy Management</h3>

        <p className="text-gray-700 text-sm">Match tasks to your energy state. Feeling drained? Do low-energy tasks. Peak energy? Tackle the hard stuff.</p>

      </div>

    </div>

  );

}
